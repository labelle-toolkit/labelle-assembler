pub fn build(b: *std.Build) !void {
    var target = b.standardTargetOptions(.{});
    const optimize = b.standardOptimizeOption(.{});

.deps
    const core_dep = b.dependency("labelle_core", .{ .target = target, .optimize = optimize });
    const core_mod = core_dep.module("labelle-core");

    const gfx_dep = b.dependency("labelle_gfx", .{ .target = target, .optimize = optimize });
    const gfx_mod = gfx_dep.module("labelle-gfx");

    const engine_dep = b.dependency("engine", .{ .target = target, .optimize = optimize });
    const engine_mod = engine_dep.module("engine");

    // Deduplicate labelle-core across gfx and engine — ensures a single core
    // module in the build, preventing diamond dependency version mismatches.
    // Use overrideImport to avoid GPA leak from addImport key re-allocation.
    overrideImport(gfx_mod, "labelle-core", core_mod);
    overrideImport(engine_mod, "labelle-core", core_mod);
    overrideImport(engine_mod, "labelle-gfx", gfx_mod);

.backend_raylib
    const backend_dep = b.dependency("labelle_raylib", .{ .target = target, .optimize = optimize });
    const backend_gfx = backend_dep.module("gfx");
    const backend_input = backend_dep.module("input");
    const backend_audio = backend_dep.module("audio");
    const backend_window = backend_dep.module("window");
    const raylib_artifact = backend_dep.artifact("raylib");

.backend_sokol
    const backend_dep = b.dependency("labelle_sokol", .{ .target = target, .optimize = optimize });
    const backend_gfx = backend_dep.module("gfx");
    const backend_input = backend_dep.module("input");
    const backend_audio = backend_dep.module("audio");
    const backend_window = backend_dep.module("window");
    const sokol_clib = backend_dep.artifact("sokol_clib");

.backend_sokol_wasm
    const backend_dep = b.dependency("labelle_sokol", .{ .target = target, .optimize = optimize });
    const backend_gfx = backend_dep.module("gfx");
    const backend_input = backend_dep.module("input");
    const backend_audio = backend_dep.module("audio");
    const backend_window = backend_dep.module("window");

.backend_sdl
    const backend_dep = b.dependency("labelle_sdl", .{ .target = target, .optimize = optimize });
    const backend_gfx = backend_dep.module("gfx");
    const backend_input = backend_dep.module("input");
    const backend_audio = backend_dep.module("audio");
    const backend_window = backend_dep.module("window");

.backend_bgfx
    const backend_dep = b.dependency("labelle_bgfx", .{ .target = target, .optimize = optimize });
    const backend_gfx = backend_dep.module("gfx");
    const backend_input = backend_dep.module("input");
    const backend_audio = backend_dep.module("audio");
    const backend_window = backend_dep.module("window");
    const bgfx_artifact = backend_dep.artifact("bgfx");
    const glfw_artifact = backend_dep.artifact("glfw");

.backend_wgpu
    const backend_dep = b.dependency("labelle_wgpu", .{ .target = target, .optimize = optimize });
    const backend_gfx = backend_dep.module("gfx");
    const backend_input = backend_dep.module("input");
    const backend_audio = backend_dep.module("audio");
    const backend_window = backend_dep.module("window");
    const glfw_artifact = backend_dep.artifact("glfw");

.ecs_adapter
    const ecs_dep = b.dependency("{{ecs_dep_name}}", .{ .target = target, .optimize = optimize });
    const ecs_mod = ecs_dep.module("ecs");

.gui_backend
    const gui_dep = b.dependency("{{gui_dep_name}}", .{ .target = target, .optimize = optimize });
    const gui_mod = gui_dep.module("{{gui_mod_name}}");

.gui_bridge
    const bridge_dep = b.dependency("gui_bridge", .{ .target = target, .optimize = optimize });
    const gui_bridge_artifact = bridge_dep.artifact("{{bridge_artifact_name}}");

.wasm_target
    // WASM/Emscripten: override target so all dependencies build for wasm32-emscripten.
    target = b.resolveTargetQuery(.{
        .cpu_arch = .wasm32,
        .os_tag = .emscripten,
    });

.exe_start
    const exe = b.addExecutable(.{
        .name = "game",
        .root_module = b.createModule(.{
            .root_source_file = b.path("main.zig"),
            .target = target,
            .optimize = optimize,
            .imports = &.{
                .{ .name = "labelle-core", .module = core_mod },
                .{ .name = "labelle-gfx", .module = gfx_mod },
                .{ .name = "labelle-engine", .module = engine_mod },
                .{ .name = "backend_gfx", .module = backend_gfx },
                .{ .name = "backend_input", .module = backend_input },
                .{ .name = "backend_audio", .module = backend_audio },
                .{ .name = "backend_window", .module = backend_window },

.exe_ecs_import
                .{ .name = "ecs_backend", .module = ecs_mod },

.exe_gui_import
                .{ .name = "gui_backend", .module = gui_mod },

.exe_end
            },
        }),
    });

.link_raylib
    exe.linkLibrary(raylib_artifact);

.wasm_emsdk_raylib
    const emsdk = @import("labelle_raylib").emsdk;

.wasm_emsdk_sokol
    const sokol_em = @import("labelle_sokol");
    const emsdk_dep = b.dependency("emsdk", .{});

.wasm_exe_start
    const wasm = b.addLibrary(.{
        .name = "game",
        .root_module = b.createModule(.{
            .root_source_file = b.path("main.zig"),
            .target = target,
            .optimize = optimize,
            .imports = &.{
                .{ .name = "labelle-core", .module = core_mod },
                .{ .name = "labelle-gfx", .module = gfx_mod },
                .{ .name = "labelle-engine", .module = engine_mod },
                .{ .name = "backend_gfx", .module = backend_gfx },
                .{ .name = "backend_input", .module = backend_input },
                .{ .name = "backend_audio", .module = backend_audio },
                .{ .name = "backend_window", .module = backend_window },

.wasm_exe_ecs_import
                .{ .name = "ecs_backend", .module = ecs_mod },

.wasm_exe_gui_import
                .{ .name = "gui_backend", .module = gui_mod },

.wasm_exe_end
            },
        }),
    });

.link_raylib_wasm
    const emcc_flags = emsdk.emccDefaultFlags(b.allocator, .{
        .optimize = optimize,
        .asyncify = true,
    });
    var emcc_settings = emsdk.emccDefaultSettings(b.allocator, .{
        .optimize = optimize,
    });
    emcc_settings.put("ALLOW_MEMORY_GROWTH", "1") catch unreachable;
    const install_dir: std.Build.InstallDir = .{ .custom = "web" };
    const emcc_step = emsdk.emccStep(b, raylib_artifact, wasm, .{
        .optimize = optimize,
        .flags = emcc_flags,
        .settings = emcc_settings,
        .install_dir = install_dir,
    });

.link_sokol_wasm
    wasm.linkLibrary(backend_dep.artifact("sokol_clib"));
    const link_step = try sokol_em.emLinkStep(b, .{
        .target = target,
        .optimize = optimize,
        .lib_main = wasm,
        .emsdk = emsdk_dep,
        .shell_file_path = null,
        .use_webgl2 = true,
        .release_use_closure = false,
    });
    const emcc_step = &link_step.step;

.wasm_footer
    b.getInstallStep().dependOn(emcc_step);
    const run_step = b.step("run", "Serve WASM build");
    run_step.dependOn(emcc_step);
}

/// Override a module import without leaking memory.
fn overrideImport(m: *std.Build.Module, name: []const u8, module: *std.Build.Module) void {
    const gop = m.import_table.getOrPut(m.owner.allocator, name) catch @panic("OOM");
    if (!gop.found_existing) {
        gop.key_ptr.* = name;
    }
    gop.value_ptr.* = module;
}

/// Get iOS SDK path using xcrun
fn getIosSdkPath(b: *std.Build, sdk_name: []const u8) ?[]const u8 {
    const result = std.process.Child.run(.{
        .allocator = b.allocator,
        .argv = &.{ "xcrun", "--sdk", sdk_name, "--show-sdk-path" },
    }) catch return null;
    defer b.allocator.free(result.stdout);
    defer b.allocator.free(result.stderr);
    if (result.term == .Exited and result.term.Exited == 0) {
        const path = std.mem.trim(u8, result.stdout, &std.ascii.whitespace);
        if (path.len == 0) return null;
        return b.allocator.dupe(u8, path) catch null;
    }
    return null;
}

/// Configure SDK paths for a compile artifact (sokol_clib)
fn configureSdkPaths(b: *std.Build, artifact: *std.Build.Step.Compile, sdk_path: []const u8) void {
    artifact.root_module.addSystemIncludePath(.{ .cwd_relative = b.pathJoin(&.{ sdk_path, "usr/include" }) });
    artifact.root_module.addSystemFrameworkPath(.{ .cwd_relative = b.pathJoin(&.{ sdk_path, "System/Library/Frameworks" }) });
    artifact.root_module.addSystemFrameworkPath(.{ .cwd_relative = b.pathJoin(&.{ sdk_path, "System/Library/SubFrameworks" }) });
}

/// Add SDK library and framework paths to an executable
fn addExeSdkPaths(b: *std.Build, exe: *std.Build.Step.Compile, sdk_path: []const u8) void {
    exe.root_module.addLibraryPath(.{ .cwd_relative = b.pathJoin(&.{ sdk_path, "usr/lib" }) });
    exe.root_module.addFrameworkPath(.{ .cwd_relative = b.pathJoin(&.{ sdk_path, "System/Library/Frameworks" }) });
}

/// Link iOS system frameworks on an executable
fn linkIosFrameworks(exe: *std.Build.Step.Compile) void {
    exe.root_module.linkFramework("Foundation", .{});
    exe.root_module.linkFramework("UIKit", .{});
    exe.root_module.linkFramework("Metal", .{});
    exe.root_module.linkFramework("MetalKit", .{});
    exe.root_module.linkFramework("AudioToolbox", .{});
    exe.root_module.linkFramework("AVFoundation", .{});
    exe.root_module.linkFramework("QuartzCore", .{});
}

pub fn build(b: *std.Build) void {
    const optimize = b.standardOptimizeOption(.{});
    const device_mode = b.option(bool, "device", "Build for iOS device instead of simulator") orelse false;

    const sdk_name: []const u8 = if (device_mode) "iphoneos" else "iphonesimulator";
    const sdk_path = getIosSdkPath(b, sdk_name) orelse
        @panic("Could not find iOS SDK. Is Xcode installed?");

    const host_arch = b.host.result.cpu.arch;
    const ios_target = if (device_mode)
        b.resolveTargetQuery(.{
            .cpu_arch = .aarch64,
            .os_tag = .ios,
        })
    else if (host_arch == .x86_64)
        b.resolveTargetQuery(.{
            .cpu_arch = .x86_64,
            .os_tag = .ios,
            .abi = .simulator,
        })
    else
        b.resolveTargetQuery(.{
            .cpu_arch = .aarch64,
            .os_tag = .ios,
            .abi = .simulator,
            .cpu_model = .{ .explicit = &std.Target.aarch64.cpu.apple_a14 },
        });

.ios_target_alias
    const target = ios_target;

.ios_deps
    const core_dep = b.dependency("labelle_core", .{ .target = ios_target, .optimize = optimize });
    const core_mod = core_dep.module("labelle-core");

    const gfx_dep = b.dependency("labelle_gfx", .{ .target = ios_target, .optimize = optimize });
    const gfx_mod = gfx_dep.module("labelle-gfx");

    const engine_dep = b.dependency("engine", .{ .target = ios_target, .optimize = optimize });
    const engine_mod = engine_dep.module("engine");

    overrideImport(gfx_mod, "labelle-core", core_mod);
    overrideImport(engine_mod, "labelle-core", core_mod);
    overrideImport(engine_mod, "labelle-gfx", gfx_mod);

.backend_sokol_ios
    // sokol with dont_link_system_libs — we link iOS frameworks manually
    const backend_dep = b.dependency("labelle_sokol", .{
        .target = ios_target,
        .optimize = optimize,
        .dont_link_system_libs = true,
    });
    const backend_gfx = backend_dep.module("gfx");
    const backend_input = backend_dep.module("input");
    const backend_audio = backend_dep.module("audio");
    const backend_window = backend_dep.module("window");
    const sokol_clib = backend_dep.artifact("sokol_clib");

    // Configure SDK paths on sokol_clib for C header compilation
    configureSdkPaths(b, sokol_clib, sdk_path);

.ios_exe_start
    const exe = b.addExecutable(.{
        .name = "game",
        .root_module = b.createModule(.{
            .root_source_file = b.path("main.zig"),
            .target = ios_target,
            .optimize = optimize,
            .imports = &.{
                .{ .name = "labelle-core", .module = core_mod },
                .{ .name = "labelle-gfx", .module = gfx_mod },
                .{ .name = "labelle-engine", .module = engine_mod },
                .{ .name = "backend_gfx", .module = backend_gfx },
                .{ .name = "backend_input", .module = backend_input },
                .{ .name = "backend_audio", .module = backend_audio },
                .{ .name = "backend_window", .module = backend_window },

.ios_exe_ecs_import
                .{ .name = "ecs_backend", .module = ecs_mod },

.ios_exe_gui_import
                .{ .name = "gui_backend", .module = gui_mod },

.ios_exe_end
            },
        }),
    });

.ios_link
    exe.linkLibrary(sokol_clib);
    exe.linkLibC();
    addExeSdkPaths(b, exe, sdk_path);
    linkIosFrameworks(exe);

.ios_link_gui_bridge
    exe.linkLibrary(gui_bridge_artifact);

.ios_footer
    b.installArtifact(exe);

    const run_step = b.step("run", "Build for iOS");
    run_step.dependOn(&exe.step);
}

/// Override a module import without leaking memory.
fn overrideImport(m: *std.Build.Module, name: []const u8, module: *std.Build.Module) void {
    const gop = m.import_table.getOrPut(m.owner.allocator, name) catch @panic("OOM");
    if (!gop.found_existing) {
        gop.key_ptr.* = name;
    }
    gop.value_ptr.* = module;
}

.footer
    b.installArtifact(exe);

    const run_cmd = b.addRunArtifact(exe);
    run_cmd.step.dependOn(b.getInstallStep());
    const run_step = b.step("run", "Run the game");
    run_step.dependOn(&run_cmd.step);
}

/// Override a module import without leaking memory.
/// Zig's addImport always calls b.dupe(name), leaking the old key on replacement
/// and creating unnecessary allocations for new keys. This function accesses the
/// import_table directly: reuses existing keys and avoids b.dupe entirely.
fn overrideImport(m: *std.Build.Module, name: []const u8, module: *std.Build.Module) void {
    const gop = m.import_table.getOrPut(m.owner.allocator, name) catch @panic("OOM");
    if (!gop.found_existing) {
        // New import — store our key (string literal from generated code, lives forever)
        gop.key_ptr.* = name;
    }
    // Replace value (existing or new)
    gop.value_ptr.* = module;
}
