var gpa = std.heap.GeneralPurposeAllocator(.{}){};

var frame_count: u32 = 0;

const screen_w: u32 = {{width}};
const screen_h: u32 = {{height}};
const screen_title = "{{title}}";

// {{platform_comment}}
export fn init() callconv(.c) void {
    window.initGfx();
    const allocator = gpa.allocator();
    g = AssembledGame.init(allocator);

    g.setHooks(&hooks);
    g.setScreenHeight(@as(f32, @floatFromInt(screen_h)));
{{init_code}}}

export fn frame() callconv(.c) void {
    frame_count +|= 1;

    const actual_w = window.width();
    const actual_h = window.height();
    BackendGfxType.setScreenSize(actual_w, actual_h);
    g.setScreenHeight(@floatFromInt(actual_h));
    const dt: f32 = 0.016;
{{tick_code}}    g.tick(dt);
    const pass_action = window.beginFrame();
    window.beginPass(pass_action);
    g.render();
    g.renderGizmos();
    window.endFrame();
}

export fn cleanup() callconv(.c) void {
    g.deinit();
    // On iOS, don't shutdown sokol_gfx — sokol_app owns the Metal device and
    // UIWindow lifecycle. On Android, shutdown is safe.
    if (comptime @import("builtin").target.os.tag != .ios) {
        window.shutdownGfx();
    }
    _ = gpa.deinit();
}

// {{entry_comment}}
pub fn main() void {
    window.run(.{
        .init_cb = &init,
        .frame_cb = &frame,
        .cleanup_cb = &cleanup,
        .w = screen_w,
        .h = screen_h,
        .title = screen_title,
    });
}
